2. Stream Manager — 1 Process ต่อ 1 กล้อง
pythonimport multiprocessing as mp
import cv2

class StreamManager:
    def __init__(self, cameras_config, motion_queue):
        self.cameras = cameras_config
        self.motion_queue = motion_queue  # ส่งต่อไป AI

    def start_all(self):
        processes = []
        for cam in self.cameras:
            p = mp.Process(
                target=self._capture_loop,
                args=(cam,)
            )
            p.start()
            processes.append(p)
        return processes

    def _capture_loop(self, cam):
        cap = cv2.VideoCapture(cam['rtsp'])
        motion_detector = MotionDetector()
        
        while True:
            ret, frame = cap.read()
            if not ret:
                continue

            # ตรวจ Motion ก่อน (เบามาก ใช้ CPU น้อย)
            if motion_detector.detect(frame):
                self.motion_queue.put({
                    "cam_id": cam['id'],
                    "zone": cam['zone'],
                    "frame": frame,
                    "timestamp": time.time()
                })

3. Motion Detector — กรองก่อนส่ง AI
pythonclass MotionDetector:
    def __init__(self, threshold=1000):
        self.bg_subtractor = cv2.createBackgroundSubtractorMOG2(
            history=500,
            varThreshold=50
        )
        self.threshold = threshold

    def detect(self, frame):
        # ลด resolution ก่อนตรวจ Motion — ประหยัด CPU
        small = cv2.resize(frame, (320, 240))
        mask = self.bg_subtractor.apply(small)
        
        # นับ pixel ที่เปลี่ยนแปลง
        motion_pixels = cv2.countNonZero(mask)
        
        return motion_pixels > self.threshold

4. Priority Queue — จัดลำดับความสำคัญ
pythonimport heapq

class PriorityEventQueue:
    PRIORITY = {
        "critical": 1,   # สูงสุด — ประมวลผลก่อน
        "normal": 2,
        "low": 3
    }
    
    def __init__(self):
        self._queue = []
        self._lock = threading.Lock()

    def put(self, event):
        priority = self.PRIORITY.get(event['zone'], 3)
        with self._lock:
            heapq.heappush(self._queue, (priority, event))

    def get(self):
        with self._lock:
            if self._queue:
                return heapq.heappop(self._queue)[1]
        return None

5. AI Engine — 2 Workers รัน Parallel
pythonfrom ultralytics import YOLO
import threading

class AIEngine:
    def __init__(self, queue, event_processor, num_workers=2):
        self.queue = queue
        self.processor = event_processor
        self.num_workers = num_workers
        
        # โหลด model 1 ครั้ง แชร์กัน (thread-safe)
        self.model = YOLO("yolo11n.onnx")

    def start(self):
        for i in range(self.num_workers):
            t = threading.Thread(
                target=self._inference_loop,
                name=f"AI-Worker-{i}"
            )
            t.daemon = True
            t.start()

    def _inference_loop(self):
        while True:
            event = self.queue.get()
            if event is None:
                continue

            # Run YOLO
            results = self.model(
                event['frame'],
                classes=[0],      # 0 = person เท่านั้น
                conf=0.5,
                imgsz=416         # เล็กลง = เร็วขึ้น
            )

            detections = results[0].boxes
            if len(detections) > 0:
                self.processor.handle(event, detections)

6. Event Processor — ป้องกันแจ้งซ้ำ
pythonclass EventProcessor:
    def __init__(self, alert_system, cooldown=30):
        self.alert = alert_system
        self.cooldown = cooldown
        self.last_alert = {}  # {cam_id: timestamp}

    def handle(self, event, detections):
        cam_id = event['cam_id']
        now = time.time()

        # Cooldown Check — ไม่แจ้งซ้ำถี่เกินไป
        if cam_id in self.last_alert:
            if now - self.last_alert[cam_id] < self.cooldown:
                return  # ข้ามไป

        self.last_alert[cam_id] = now
        
        # ส่งแจ้งเตือน
        count = len(detections)
        self.alert.send(
            image=event['frame'],
            message=f"⚠️ พบคน {count} คน | กล้อง {cam_id}"
        )

💡 เทคนิคสำคัญ — Shared Memory
ปัญหาของการส่งเฟรม:
Process A (AI) ──copy──► Process B (Web) = RAM เพิ่มเท่าตัว ❌

แก้ด้วย Shared Memory:
Process A เขียน ──► [ช่องหน่วยความจำกลาง] ◄── Process B อ่าน
                      ไม่มีการ copy เลย ✅
pythonimport multiprocessing.shared_memory as shm
import numpy as np

class FrameBuffer:
    """
    เก็บเฟรมล่าสุดของแต่ละกล้องใน Shared Memory
    ทุก Process อ่านได้โดยไม่ต้อง copy
    """
    def __init__(self, cam_count=10):
        self.buffers = {}
        # 480p frame = 480×640×3 bytes = ~900KB ต่อกล้อง
        # 10 กล้อง = ~9MB เท่านั้น
        frame_size = 480 * 640 * 3

        for i in range(cam_count):
            buf = shm.SharedMemory(
                name=f"cam_{i}",
                create=True,
                size=frame_size
            )
            self.buffers[i] = buf

    def write(self, cam_id, frame):
        """AI Engine เขียนเฟรมที่ผ่าน YOLO แล้ว"""
        resized = cv2.resize(frame, (640, 480))
        buf = self.buffers[cam_id]
        arr = np.ndarray((480, 640, 3),
                         dtype=np.uint8,
                         buffer=buf.buf)
        arr[:] = resized

    def read(self, cam_id):
        """Web Server อ่านเฟรมล่าสุด"""
        buf = self.buffers[cam_id]
        arr = np.ndarray((480, 640, 3),
                         dtype=np.uint8,
                         buffer=buf.buf)
        return arr.copy()

pythonfrom fastapi import FastAPI, WebSocket
from fastapi.responses import StreamingResponse
import asyncio, cv2, base64

app = FastAPI()
frame_buffer = FrameBuffer()

# ── Option 1: WebSocket (แนะนำ) ──────────────────────
@app.websocket("/ws/cam/{cam_id}")
async def camera_websocket(websocket: WebSocket, cam_id: int):
    await websocket.accept()
    
    try:
        while True:
            frame = frame_buffer.read(cam_id)
            
            # Encode เป็น JPEG ก่อนส่ง (ลดขนาด 10x)
            _, jpg = cv2.imencode(
                '.jpg', frame,
                [cv2.IMWRITE_JPEG_QUALITY, 70]  # 70% quality
            )
            
            # ส่งเป็น base64
            b64 = base64.b64encode(jpg).decode()
            await websocket.send_text(b64)
            
            # 5 FPS = sleep 200ms
            await asyncio.sleep(0.2)
            
    except Exception:
        await websocket.close()


# ── Option 2: MJPEG Stream (fallback) ────────────────
@app.get("/stream/cam/{cam_id}")
async def mjpeg_stream(cam_id: int):
    def generate():
        while True:
            frame = frame_buffer.read(cam_id)
            _, jpg = cv2.imencode('.jpg', frame,
                       [cv2.IMWRITE_JPEG_QUALITY, 70])
            yield (
                b"--frame\r\n"
                b"Content-Type: image/jpeg\r\n\r\n"
                + jpg.tobytes()
                + b"\r\n"
            )
            time.sleep(0.2)
    
    return StreamingResponse(
        generate(),
        media_type="multipart/x-mixed-replace;boundary=frame"
    )

💡 Core Design Principle
หัวใจสำคัญคือ แยก Pipeline ออกจากกัน
❌ แบบผิด — ทำทุกอย่างตามลำดับ (Sequential)
Capture → รอ AI → รอ Encode → แสดงผล
ผลลัพธ์ : ถ้า AI ช้า ทุกอย่างหยุดรอ = แลค

✅ แบบถูก — ทำพร้อมกัน (Parallel Pipeline)
Capture    ──► ทำงานตลอดเวลา ไม่รอใคร
     ↓
Display    ──► แสดงเฟรมล่าสุดที่มีเสมอ ไม่รอ AI
     ↑
AI Engine  ──► ประมวลผลแล้ว update ข้อมูล overlay
แนวคิดคือ Display ไม่รอ AI เลย
แสดงวิดีโอสดตลอดเวลา แล้ว AI ค่อย overlay ข้อมูลทับ

Framework ที่คุณเสนอ
Alert = Object + Behavior + Zone + Time

Object   → ใคร/อะไร
Behavior → ทำอะไร
Zone     → ที่ไหน
Time     → เมื่อไหร่
นี่คือ Rule Engine ครับ ผู้ใช้กำหนด Rule เอง แล้วระบบแจ้งเตือนเมื่อ Rule ตรงกัน

ขยาย Use Cases จาก Framework นี้
Object
คน
├── คนทั่วไป
├── คนที่ได้รับอนุญาต (ถ้ามี Face Recognition)
└── คนที่ไม่ได้รับอนุญาต

สัตว์
├── สุนัข / แมว
├── สัตว์ขนาดใหญ่
└── สัตว์ที่ไม่พึงประสงค์

ยานพาหนะ
├── รถยนต์
├── รถจักรยานยนต์
└── รถที่ไม่รู้จัก (ถ้ามีระบบทะเบียน)

สิ่งของ
├── วัตถุต้องสงสัย
└── สิ่งของที่วางทิ้งไว้นานผิดปกติ

Behavior
การบุกรุก
├── เข้าพื้นที่โดยไม่ได้รับอนุญาต
├── ข้ามเส้นเขตแดน (Line Crossing)
└── เข้าในช่วงเวลาห้าม

พฤติกรรมน่าสงสัย
├── อยู่ในพื้นที่นานผิดปกติ (Loitering)
├── เดินวนซ้ำหลายรอบ
├── วิ่ง / เคลื่อนไหวเร็วผิดปกติ
└── นอนหมอบ / ซ่อนตัว

เหตุการณ์เฉพาะ
├── ทิ้งสิ่งของแล้วจากไป
├── รวมกลุ่มคนจำนวนมากผิดปกติ
└── ล้มหรือหกล้ม (Fall Detection)

Zone
กำหนดได้หลายรูปแบบ
├── Virtual Line     → ขีดเส้น ถ้าข้ามแจ้งเตือน
├── Polygon Zone     → วาดพื้นที่ไม่ให้เข้า
├── Direction Zone   → เข้าได้ทางเดียว ถ้าย้อนทิศแจ้งเตือน
└── Full Frame Zone  → ทั้งภาพกล้อง

Time
├── ช่วงเวลากลางคืน เช่น 20:00 - 06:00
├── นอกเวลาทำการ เช่น หลัง 18:00 วันธรรมดา
├── วันหยุด / วันเสาร์อาทิตย์
└── กำหนดเองได้อิสระ

Pain Points ที่ต้องออกแบบให้หลีกเลี่ยง
❌ แจ้งเตือนบ่อยเกินไป
   User จะเริ่มเพิกเฉย และปิดการแจ้งเตือน
   แก้ด้วย Cooldown + Sensitivity ปรับได้

❌ ระบบล่มกลางดึก
   เหตุการณ์เกิดแต่ไม่มีใครรู้
   แก้ด้วย Auto-restart + แจ้งเตือนเมื่อระบบมีปัญหา

❌ หากล้องที่แจ้งเตือนไม่เจอ
   User ตื่นตระหนกแต่หาข้อมูลไม่ได้
   แก้ด้วย Alert ที่ระบุกล้อง + Snapshot ชัดเจน

❌ ตั้งค่ายากเกินไป
   Admin ตั้ง Rule ไม่เป็น ระบบก็ไม่ทำงาน
   แก้ด้วย UI ที่ใช้ง่าย มี Template Rule สำเร็จรูป

❌ วิดีโอย้อนหลังหาไม่เจอ
   เกิดเหตุแล้วหาหลักฐานไม่ได้
   แก้ด้วย Event-based Recording + ค้นหาง่าย

         ↓

   ผลลัพธ์ : Person, confidence 0.97
         ↓
4. Rule Engine เช็ค
   ตรง Rule #2 "คนน่าสงสัยกลางคืน"
         ↓
5. Alert Manager
   ├── บันทึก Event ลง PostgreSQL
   ├── บันทึก Snapshot ลง File System
   ├── ส่ง LINE Notify
   └── Push WebSocket ไป Frontend
         ↓
6. Frontend
   ├── กรอบ CAM 3 เปลี่ยนเป็นสีแดง
   ├── Badge แจ้งเตือนปรากฏ

Data Flow — User ถามคำถาม
1. User พิมพ์คำถามภาษาไทย
         ↓

         ↓

         ↓
4. Validate SQL
   ├── SELECT เท่านั้น

         ↓
5. Query PostgreSQL
         ↓
6. คืนผลลัพธ์ + Snapshot
   กลับไปแสดงที่ Frontend
